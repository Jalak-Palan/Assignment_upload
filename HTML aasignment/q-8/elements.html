Introduction:

int main() {
    int arr[] = {12, 45, 7, 23, 19};
    int n = 5, key = 23, found = -1;

    for (int i = 0; i < n; i++) {
        if (arr[i] == key) {
            found = i;   // index where found
            break;
        }
    }

    if (found != -1)

    return 0;
}
